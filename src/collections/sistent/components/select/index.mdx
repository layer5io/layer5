import { useState } from "react";
import { FormControl, InputLabel, MenuItem, Select, OutlinedInput, Box } from "@sistent/sistent";

export const BasicSelectDemo = ({ variant, ...props }) => {
  const [age, setAge] = useState("");
  const handleChange = (event) => setAge(event.target.value);
  const labelId = `demo-select-${variant || 'outlined'}-label`;
  return (
    <FormControl sx={{ width: "200px" }} variant={variant} {...props}>
      <InputLabel id={labelId}>Age</InputLabel>
      <Select MenuProps={{ disableScrollLock: true, marginThreshold: null }}
        labelId={labelId}
        id={`demo-select-${variant || 'outlined'}`}
        value={age}
        label="Age"
        onChange={handleChange}
      >
        <MenuItem value={10}>Ten</MenuItem>
        <MenuItem value={20}>Twenty</MenuItem>
        <MenuItem value={30}>Thirty</MenuItem>
      </Select>
    </FormControl>
  );
};

export const MultipleSelectDemo = () => {
  const [names, setNames] = useState([]);
  const handleChange = (event) => {
    const { target: { value } } = event;
    setNames(typeof value === 'string' ? value.split(',') : value);
  };
  return (
    <FormControl sx={{ width: "200px" }}>
      <InputLabel id="demo-multiple-name-label">Age</InputLabel>
      <Select MenuProps={{ disableScrollLock: true, marginThreshold: null }}
        labelId="demo-multiple-name-label"
        id="demo-multiple-name"
        multiple
        value={names}
        onChange={handleChange}
        label="Age"
        input={<OutlinedInput label="Name" />}
      >
        <MenuItem value={10}>Ten</MenuItem>
        <MenuItem value={20}>Twenty</MenuItem>
        <MenuItem value={30}>Thirty</MenuItem>
      </Select>
    </FormControl>
  );
};

The Select component is a versatile and customizable element commonly used in forms and user interfaces. It is designed to present a list of options for users to choose from, either as a single selection or multiple selections. This component plays a vital role in collecting user-provided information efficiently and intuitively.
