import { useState } from "react";
import { FormHelperText, Grid } from "@mui/material";
import { FormControl, InputLabel, MenuItem, Select, ListSubheader, OutlinedInput, Box } from "@sistent/sistent";

export const StateSelectDemo = (props) => {
  const [age, setAge] = useState("");
  const handleChange = (event) => setAge(event.target.value);
  return (
    <FormControl sx={{ width: "200px" }} {...props}>
      <InputLabel id="demo-simple-select-state-label">Age</InputLabel>
      <Select MenuProps={{ disableScrollLock: true, marginThreshold: null }}
        labelId="demo-simple-select-state-label"
        id="demo-simple-select-state"
        value={age}
        label="Age"
        onChange={handleChange}
        {...props}
      >
        <MenuItem value={10}>Ten</MenuItem>
        <MenuItem value={20}>Twenty</MenuItem>
        <MenuItem value={30}>Thirty</MenuItem>
      </Select>
    </FormControl>
  );
};

export const PlaceholderSelectDemo = () => {
  const [age, setAge] = useState("");

export const MultipleSelectDemo = () => {
  const [names, setNames] = useState([]);
  const handleChange = (event) => {
    const { target: { value } } = event;
    setNames(typeof value === 'string' ? value.split(',') : value);
  };
  return (
    <FormControl sx={{ width: "200px" }}>
      <InputLabel id="demo-multiple-name-label">Age</InputLabel>
      <Select MenuProps={{ disableScrollLock: true, marginThreshold: null }}
        multiple
        labelId="demo-multiple-name-label"
        id="demo-multiple-name"
        value={names}
        onChange={handleChange}
        label="Age"
        input={<OutlinedInput label="Name" />}
      >
        <MenuItem value={10}>Ten</MenuItem>
        <MenuItem value={20}>Twenty</MenuItem>
        <MenuItem value={30}>Thirty</MenuItem>
      </Select>
    </FormControl>
  );
};
