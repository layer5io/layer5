import { useState } from "react";
import { ClickAwayListener, Button, Box } from "@sistent/sistent";

export const ClickAwayListenerCodeAll = () => {
  const [openBasic, setOpenBasic] = useState(false);
  const [openCustom, setOpenCustom] = useState(false);
  const [openDismissible, setOpenDismissible] = useState(false);

  return (
    <ThemeWrapper>

      <a id="Basic ClickAwayListener"><h2>Basic ClickAwayListener</h2></a>
      <p>ClickAwayListener monitors clicks and touches outside the wrapped element and calls the <code>onClickAway</code> callback when that happens.</p>

      <div className="showcase">
        <div className="items">
          <ClickAwayListener onClickAway={() => setOpenBasic(false)}>
            <div style={{ display: "inline-flex", flexDirection: "column", gap: "1rem" }}>
              <Button variant="contained" onClick={() => setOpenBasic((prev) => !prev)}>
                {openBasic ? "Close panel" : "Open panel"}
              </Button>
              {openBasic && (
                <Box sx={{ p: 2, bgcolor: "background.paper", border: "1px solid #d1d5db", borderRadius: "0.75rem" }}>
                  <p style={{ margin: 0 }}>Clicking outside this box will close the panel.</p>
                </Box>
              )}
            </div>
          </ClickAwayListener>
        </div>
        <CodeBlock name="basic-clickaway-listener" collapsible code={`import { useState } from "react";
import { ClickAwayListener, Button, Box } from "@sistent/sistent";

const [open, setOpen] = useState(false);

<ClickAwayListener onClickAway={() => setOpen(false)}>
  <div style={{ display: "inline-flex", flexDirection: "column", gap: "1rem" }}>
    <Button variant="contained" onClick={() => setOpen((prev) => !prev)}>
      {open ? "Close panel" : "Open panel"}
    </Button>
    {open && (
      <Box sx={{ p: 2, bgcolor: "background.paper", border: "1px solid #d1d5db", borderRadius: "0.75rem" }}>
        <p style={{ margin: 0 }}>Clicking outside this box will close the panel.</p>
      </Box>
    )}
  </div>
</ClickAwayListener>`} />
      </div>

      <a id="Custom events"><h2>Custom events</h2></a>
      <p>You can tailor when the click-away behavior triggers by configuring the <code>mouseEvent</code> and <code>touchEvent</code> props.</p>

      <div className="showcase">
        <div className="items">
          <ClickAwayListener onClickAway={() => setOpenCustom(false)}>
            <div style={{ display: "inline-flex", flexDirection: "column", gap: "1rem" }}>
              <Button variant="outlined" onClick={() => setOpenCustom((prev) => !prev)}>
                {openCustom ? "Close panel" : "Open panel"}
              </Button>
              {openCustom && (
                <Box sx={{ p: 2, bgcolor: "background.paper", border: "1px solid #d1d5db", borderRadius: "0.75rem" }}>
                  <p style={{ margin: 0 }}>Click anywhere outside this box to close the panel.</p>
                </Box>
              )}
            </div>
          </ClickAwayListener>
        </div>
        <CodeBlock name="clickaway-custom-events" collapsible code={`import { useState } from "react";
import { ClickAwayListener, Button, Box } from "@sistent/sistent";

const [open, setOpen] = useState(false);

<ClickAwayListener
  onClickAway={() => setOpen(false)}
  mouseEvent="onMouseDown"
  touchEvent="onTouchEnd"
>
  <div style={{ display: "inline-flex", flexDirection: "column", gap: "1rem" }}>
    <Button variant="outlined" onClick={() => setOpen((prev) => !prev)}>
      {open ? "Close panel" : "Open panel"}
    </Button>
    {open && (
      <Box sx={{ p: 2, bgcolor: "background.paper", border: "1px solid #d1d5db", borderRadius: "0.75rem" }}>
        <p style={{ margin: 0 }}>Click anywhere outside this box to close the panel.</p>
      </Box>
    )}
  </div>
</ClickAwayListener>`} />
      </div>

      <a id="Dismissible UI patterns"><h2>Dismissible UI patterns</h2></a>
      <p>ClickAwayListener is most useful for overlay patterns like dropdowns, popovers, and menus. Keep all interactive child elements inside the wrapper to avoid unintentional dismissal.</p>

      <div className="showcase">
        <div className="items">
          <ClickAwayListener onClickAway={() => setOpenDismissible(false)}>
            <div style={{ position: "relative", display: "inline-block" }}>
              <Button variant="contained" onClick={() => setOpenDismissible((prev) => !prev)}>
                Toggle menu
              </Button>
              {openDismissible && (
                <Box sx={{ position: "absolute", top: "48px", left: 0, bgcolor: "background.paper", border: "1px solid #d1d5db", borderRadius: "0.75rem", p: 1, minWidth: "160px", zIndex: 10 }}>
                  <p style={{ margin: "0.25rem 0", padding: "0.25rem 0.5rem", cursor: "pointer" }} onClick={() => setOpenDismissible(false)}>Profile</p>
                  <p style={{ margin: "0.25rem 0", padding: "0.25rem 0.5rem", cursor: "pointer" }} onClick={() => setOpenDismissible(false)}>Settings</p>
                  <p style={{ margin: "0.25rem 0", padding: "0.25rem 0.5rem", cursor: "pointer" }} onClick={() => setOpenDismissible(false)}>Logout</p>
                </Box>
              )}
            </div>
          </ClickAwayListener>
        </div>
        <CodeBlock name="clickaway-patterns" collapsible code={`import { useState } from "react";
import { ClickAwayListener, Button, Box } from "@sistent/sistent";

  <div style={{ position: "relative", display: "inline-block" }}>
    <Button variant="contained" onClick={() => setOpen((prev) => !prev)}>
      Toggle menu
    </Button>
    {open && (
      <Box sx={{ position: "absolute", top: "48px", left: 0, bgcolor: "background.paper", border: "1px solid #d1d5db", borderRadius: "0.75rem", p: 1, minWidth: "160px", zIndex: 10 }}>
        <p>Profile</p>
        <p>Settings</p>
        <p>Logout</p>
      </Box>
    )}
  </div>
</ClickAwayListener>`} />
      </div>

    </ThemeWrapper>
  );
};
