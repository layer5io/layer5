Menus provide contextual actions and options to users in a compact, organized manner. Below are various examples demonstrating different menu configurations and use cases.

<a id="Basic Menu">
  <h2>Basic Menu</h2>
</a>

A basic menu displays a simple list of menu items anchored to a button or element.

<h3>Simple Menu</h3>

The most straightforward implementation of a menu with a text button and simple menu items.

export const SimpleMenuDemo = () => {
  const [anchorEl, setAnchorEl] = useState(null);
  const open = Boolean(anchorEl);
  const handleClick = (event) => {
    setAnchorEl(event.currentTarget);
  };
  const handleClose = () => {
    setAnchorEl(null);
  };
  return (
    <>
      <Button
        variant="contained"
        onClick={handleClick}
        label="Open Menu"
      />
      <Menu
        id="simple-menu"
        anchorEl={anchorEl}
        open={open}
        onClose={handleClose}
      >
        <MenuItem onClick={handleClose}>Profile</MenuItem>
        <MenuItem onClick={handleClose}>My account</MenuItem>
        <MenuItem onClick={handleClose}>Logout</MenuItem>
      </Menu>
    </>
  );
};
