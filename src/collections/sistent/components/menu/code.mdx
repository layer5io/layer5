import { useState } from "react";
import { Box, Button, Divider, Menu, MenuItem, MenuList, Paper } from "@sistent/sistent";

export const BasicMenuCodeDemo = () => {
  const [anchorEl, setAnchorEl] = useState(null);
  const open = Boolean(anchorEl);
  const handleClick = (event) => setAnchorEl(event.currentTarget);
  const handleClose = () => setAnchorEl(null);

  return (
    <Box sx={{ display: "flex", justifyContent: "center", width: "100%" }}>
      <Button variant="contained" label="Open Menu" onClick={handleClick} />
      <Menu
        anchorEl={anchorEl}
        open={open}
        onClose={handleClose}
        disableScrollLock
        marginThreshold={null}
      >
        <MenuItem onClick={handleClose}>Profile</MenuItem>
        <MenuItem onClick={handleClose}>Account</MenuItem>
        <MenuItem onClick={handleClose}>Sign out</MenuItem>
      </Menu>
    </Box>
  );
};

export const PositionedMenuCodeDemo = () => {
  const [anchorEl, setAnchorEl] = useState(null);
  const open = Boolean(anchorEl);
  const handleClick = (event) => setAnchorEl(event.currentTarget);
  const handleClose = () => setAnchorEl(null);

  return (
    <Box sx={{ display: "flex", justifyContent: "center", width: "100%" }}>
      <Button variant="outlined" label="Project Actions" onClick={handleClick} />
      <Menu
        anchorEl={anchorEl}
        open={open}
        onClose={handleClose}
        anchorOrigin={{ vertical: "bottom", horizontal: "right" }}
        transformOrigin={{ vertical: "top", horizontal: "right" }}
        disableScrollLock
        marginThreshold={null}
      >
        <MenuItem onClick={handleClose}>Rename</MenuItem>
        <MenuItem onClick={handleClose}>Duplicate</MenuItem>
        <Divider />
        <MenuItem onClick={handleClose}>Archive</MenuItem>
      </Menu>
    </Box>
  );
};
