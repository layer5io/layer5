import { Collapse, Button } from "@sistent/sistent";
import { useState } from "react";

export const BasicCollapseDemo = () => {
  const [expanded, setExpanded] = useState(false);
  return (
    <>
      <Button variant="contained" onClick={() => setExpanded(!expanded)}>
        {expanded ? "Collapse" : "Expand"}
      </Button>
      <Collapse in={expanded}>
        <div>Content to be collapsed/expanded</div>
      </Collapse>
    </>
  );
};

export const DefaultExpandedCollapseDemo = () => {
  const [expanded, setExpanded] = useState(true);
  return (
    <>
      <Button variant="outlined" onClick={() => setExpanded(!expanded)}>
        {expanded ? "Collapse" : "Expand"}
      </Button>
      <Collapse in={expanded}>
        <div>Content that's expanded by default</div>
      </Collapse>
    </>
  );
};

export const UnmountOnExitCollapseDemo = () => {
  const [expanded, setExpanded] = useState(false);
  return (
    <>
      <Button variant="text" onClick={() => setExpanded(!expanded)}>
        {expanded ? "Collapse" : "Expand"}
      </Button>
      <Collapse in={expanded} unmountOnExit>
        <div>Content that unmounts when collapsed</div>
      </Collapse>
    </>
  );
};

export const CustomTransitionCollapseDemo = () => {
  const [expanded, setExpanded] = useState(false);
  return (
    <>
      <Button variant="contained" onClick={() => setExpanded(!expanded)}>
        {expanded ? "Fast Collapse" : "Slow Expand"}
      </Button>
      <Collapse 
        in={expanded} 
        timeout={expanded ? 1000 : 300} 
        easing={expanded ? 'cubic-bezier(0.4, 0, 0.2, 1)' : 'ease-out'}
      >
        <div>Custom transition content</div>
      </Collapse>
    </>
  );
};
