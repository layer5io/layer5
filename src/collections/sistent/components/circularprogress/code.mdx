title: Circular Progress Code
component: circularprogress
description: Circular progress indicators provide visual feedback about ongoing processes, helping users understand that their request is being processed. They can show either determinate progress (specific completion percentage) or indeterminate progress (unknown duration).
---

import { CircularProgress } from "@sistent/sistent";
import { useState, useEffect } from "react";

export const CircularProgressDemo = () => {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    const timer = setInterval(() => {
      setProgress((prevProgress) => (prevProgress >= 100 ? 0 : prevProgress + 10));
    }, 800);

    return () => {
      clearInterval(timer);
    };
  }, []);

  return <CircularProgress variant="determinate" value={progress} />;
};


The `CircularProgress` component is essential for communicating system status to users. It provides immediate feedback during loading states, file uploads, data processing, and other time-consuming operations.
