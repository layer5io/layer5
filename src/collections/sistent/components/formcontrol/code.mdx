import { useState } from "react";
import { FormControl, InputLabel, MenuItem, Select, TextField, Box } from "@sistent/sistent";

export const BasicFormControlCodeDemo = () => {
  return (
    <FormControl sx={{ width: "200px" }}>
      <InputLabel id="demo-form-control-label">Name</InputLabel>
      <Select
        labelId="demo-form-control-label"
        id="demo-form-control"
        label="Name"
        MenuProps={{ disableScrollLock: true, marginThreshold: null }}
      >
        <MenuItem value="john">John</MenuItem>
        <MenuItem value="jane">Jane</MenuItem>
        <MenuItem value="bob">Bob</MenuItem>
      </Select>
    </FormControl>
  );
};
